local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")
local CoreGui = game:GetService("CoreGui") or game:GetService("Players").LocalPlayer:WaitForChild("PlayerGui")

local hooked = false

local writeu32 = buffer.writeu32
local targetPacketId = 0x1B
local targetValue = 0xFFFFFFFF

local function rakhook(packet)
    if packet.PacketId == targetPacketId then
        local buf = packet.AsBuffer
        writeu32(buf, 1, targetValue)
        packet:SetData(buf)
    end
end

UICorner_Btn.CornerRadius = UDim.new(0, 8)
UICorner_Btn.Parent = ToggleBtn

local dragging, dragInput, dragStart, startPos

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

local function setHookState(state)
    hooked = state
    
    local targetColor, targetText
    if hooked then
        raknet.add_send_hook(rakhook)
        targetColor = Color3.fromRGB(40, 200, 100)
        targetText = "ON"
    else
        raknet.remove_send_hook(rakhook)
        targetColor = Color3.fromRGB(255, 60, 60)
        targetText = "OFF"
    end

    ToggleBtn.Text = targetText
    local tweenInfo = TweenInfo.new(0.3, Enum.EasingStyle.Quint, Enum.EasingDirection.Out)
    local tween = TweenService:Create(ToggleBtn, tweenInfo, {BackgroundColor3 = targetColor})
    tween:Play()
end

ToggleBtn.MouseButton1Click:Connect(function()
    setHookState(not hooked)
end)
